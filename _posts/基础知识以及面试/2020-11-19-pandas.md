# 按照pid合并后面的数据：
```
def ab(df):
    data = [eval(i) for i in df.values.tolist()]
    res = []
    for i in data:
        res.extend(i)
    data = list(set(res))
    return " ".join('%s' %(id) for id in data)

df = pd.read_csv('eng.txt', sep='\t')
df = df.groupby(['pid'])['eng'].apply(ab)
df = df.reset_index()
```
